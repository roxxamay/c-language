find the area and volume of prism?
********************************************************************/

#include"main.h"

int main()
{
    int area,v,s,a,b,c,h;

    printf("\nENTER THE 1ST SIDE OF TRAINGULAR PRISM:");
    scanf("%i",&a);

    printf("\nENTER THE 2nd SIDE OF TRAINGULAR PRISM:");
    scanf("%i",&b);

    printf("\nENTER THE 3rd SIDE OF TRAINGULAR PRISM:");
    scanf("%i",&c);

    printf("\nENTER THE HEIGHT OF TRAINGULAR PRISM:");
    scanf("%i",&h);
    //CALCULATION

    s=(a+b+c)/2.0;
    v=s*sqrt(s*(s-a)*(s-b)*(s-c))*h;
    area=((2*s*sqrt(s*(s-a)*(s-b)*(s-c)))+((a+b+c)*h));
    
    printf("\nHENCE THE REQUIRED AREA:  %i",area);
    printf("\nHENCE THE REQUIRED VOLUME:  %i",v);
    getch();

}

//program for clock
#include<stdio.h>
#include<windows.h>

int main()
{
    int s,m,h;
    int d=1000;//we add delay to program to look more realastic
    printf("ENTER THE TIME (in the format of HH:MM:SS)");
    scanf("%d:%d:%d",&h,&m,&s);
    if(h>12||m>60||s>60)
    {
        printf("\nWTF! DUDE");
        exit(0);
    }
    while(1)//this makes loop infinite
    {
        system("cls");
        s++;
        if(s>59)
        {
            m++;
            s=0;
        }
        if(m>59)
        {
            h++;
            m=0;
        }
        if(h>12)
        {
            h=1;
        }

        printf("\nCLOCK:");
        printf("%02d:%02d:%02d",h,m,s);
        Sleep(d);//1000miliseconds=1seconds
    }
}
